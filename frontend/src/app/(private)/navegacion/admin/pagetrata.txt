"use client";

import React, { useState, Suspense } from "react";
import { useRouter, useSearchParams } from "next/navigation";
import Link from "next/link";
import { 
  ArrowLeft, Pill, Calendar, Clock, Sparkles, 
  Info, Hash, CheckCircle2, Activity, CalendarDays
} from "lucide-react";

function NuevoTratamientoContent() {
  const router = useRouter();
  const searchParams = useSearchParams();
  const patientId = searchParams.get('id');

  const [isLoading, setIsLoading] = useState(false);
  const [formData, setFormData] = useState({
    nombre_esquema: "",
    tipo: "QUIMIOTERAPIA",
    fecha_inicio: "",
    frecuencia_dias: 21,
    numero_ciclos: 4,
    observaciones: ""
  });

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement>) => {
    const { name, value } = e.target;
    setFormData(prev => ({ ...prev, [name]: value }));
  };

  // Calculamos la fecha final estimada para el preview
  const fechaFinEstimada = () => {
    if (!formData.fecha_inicio) return "---";
    const fecha = new Date(formData.fecha_inicio);
    const diasTotal = formData.frecuencia_dias * formData.numero_ciclos;
    fecha.setDate(fecha.getDate() + diasTotal);
    return fecha.toLocaleDateString('es-CO', { month: 'short', day: 'numeric', year: 'numeric' });
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    console.log("Generando esquema:", formData);
    setTimeout(() => {
      setIsLoading(false);
      router.push(`/navegacion/admin/pacientes/perfil?id=${patientId}`);
    }, 1500);
  };

  return (
    <div className="w-full min-h-screen bg-slate-50/50 pb-24 p-6 md:p-10 font-sans text-slate-800">
      
      <div className="max-w-6xl mx-auto">
        {/* HEADER DE NAVEGACIÓN */}
        <div className="flex items-center gap-4 mb-8">
            <Link 
                href={`/navegacion/admin/pacientes/perfil?id=${patientId}`} 
                className="w-10 h-10 flex items-center justify-center bg-white border border-slate-200 rounded-xl text-slate-400 hover:text-blue-600 hover:border-blue-200 hover:shadow-md transition-all duration-300"
            >
                <ArrowLeft size={20} strokeWidth={2.5}/>
            </Link>
            <div>
                <h1 className="text-2xl font-black text-slate-900 tracking-tight flex items-center gap-2">
                    Nuevo Esquema Clínico
                </h1>
                <p className="text-slate-500 font-medium text-sm">Configuración del protocolo y proyección de ciclos.</p>
            </div>
        </div>

        <div className="grid grid-cols-1 lg:grid-cols-12 gap-8 items-start">
            
            {/* COLUMNA IZQUIERDA: FORMULARIO (8 cols) */}
            <div className="lg:col-span-8">
                <div className="bg-white rounded-[2rem] shadow-xl shadow-slate-200/60 p-8 border border-slate-100">
                    
                    {/* Banner Asistente IA */}
                    <div className="bg-gradient-to-r from-blue-50 to-indigo-50 border border-blue-100 p-5 rounded-2xl flex items-start gap-4 mb-8">
                        <div className="bg-white p-2.5 rounded-xl shadow-sm text-blue-600">
                            <Sparkles size={20} fill="currentColor" className="animate-pulse"/>
                        </div>
                        <div>
                            <h4 className="font-bold text-blue-900 text-sm flex items-center gap-2">
                                Auto-Planificador Inteligente
                                <span className="bg-blue-600 text-white text-[10px] px-2 py-0.5 rounded-full font-black uppercase tracking-wider">Beta</span>
                            </h4>
                            <p className="text-xs text-blue-700/80 mt-1 leading-relaxed font-medium">
                                El sistema proyectará automáticamente las fechas de los <strong>{formData.numero_ciclos} ciclos</strong> basándose en la frecuencia configurada. Podrás editar fechas individuales después si es necesario.
                            </p>
                        </div>
                    </div>

                    <form onSubmit={handleSubmit} className="space-y-8">
                        
                        {/* SECCIÓN 1: DATOS GENERALES */}
                        <div>
                            <h3 className="text-xs font-black text-slate-400 uppercase tracking-widest mb-4 flex items-center gap-2">
                                <Info size={14}/> Datos del Protocolo
                            </h3>
                            <div className="grid gap-6">
                                <div className="group">
                                    <label className="label-modern">Nombre del Esquema / Protocolo</label>
                                    <input 
                                        type="text" 
                                        name="nombre_esquema" 
                                        value={formData.nombre_esquema} 
                                        onChange={handleChange}
                                        placeholder="Ej: FOLFOX, AC-T, Esquema de Mama..." 
                                        className="input-modern"
                                        required
                                        autoFocus
                                    />
                                </div>
                                <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                                    <div className="relative group">
                                        <label className="label-modern">Tipo de Tratamiento</label>
                                        <div className="relative">
                                            <div className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400 group-focus-within:text-blue-500 transition-colors">
                                                <Activity size={18}/>
                                            </div>
                                            <select 
                                                name="tipo" 
                                                value={formData.tipo} 
                                                onChange={handleChange} 
                                                className="input-modern pl-12 appearance-none"
                                            >
                                                <option value="QUIMIOTERAPIA">Quimioterapia</option>
                                                <option value="RADIOTERAPIA">Radioterapia</option>
                                                <option value="INMUNOTERAPIA">Inmunoterapia</option>
                                                <option value="HORMONOTERAPIA">Hormonoterapia</option>
                                            </select>
                                            {/* Custom Arrow */}
                                            <div className="absolute right-4 top-1/2 -translate-y-1/2 pointer-events-none text-slate-400">
                                                <svg width="10" height="6" viewBox="0 0 10 6" fill="none" xmlns="http://www.w3.org/2000/svg"><path d="M1 1L5 5L9 1" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"/></svg>
                                            </div>
                                        </div>
                                    </div>
                                    <div className="relative group">
                                        <label className="label-modern">Fecha Inicio (Ciclo 1)</label>
                                        <div className="relative">
                                            <div className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400 group-focus-within:text-blue-500 transition-colors">
                                                <Calendar size={18}/>
                                            </div>
                                            <input 
                                                type="date" 
                                                name="fecha_inicio" 
                                                value={formData.fecha_inicio} 
                                                onChange={handleChange} 
                                                className="input-modern pl-12"
                                                required
                                            />
                                        </div>
                                    </div>
                                </div>
                            </div>
                        </div>

                        <hr className="border-slate-100"/>

                        {/* SECCIÓN 2: PARAMETRIZACIÓN */}
                        <div>
                            <h3 className="text-xs font-black text-slate-400 uppercase tracking-widest mb-4 flex items-center gap-2">
                                <Clock size={14}/> Configuración Temporal
                            </h3>
                            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                                <div className="bg-slate-50 p-4 rounded-2xl border border-slate-100 group focus-within:ring-2 focus-within:ring-blue-500/20 focus-within:border-blue-500 transition-all">
                                    <label className="label-modern !mb-1">Frecuencia (Días)</label>
                                    <div className="flex items-center gap-3">
                                        <div className="bg-white p-2 rounded-lg text-slate-400 shadow-sm">
                                            <Clock size={20}/>
                                        </div>
                                        <input 
                                            type="number" 
                                            name="frecuencia_dias" 
                                            value={formData.frecuencia_dias} 
                                            onChange={handleChange} 
                                            className="bg-transparent border-none text-xl font-bold text-slate-800 w-full focus:ring-0 p-0 placeholder:text-slate-300"
                                            min="1"
                                            placeholder="21"
                                        />
                                        <span className="text-xs font-bold text-slate-400 uppercase pr-2">Días</span>
                                    </div>
                                </div>

                                <div className="bg-slate-50 p-4 rounded-2xl border border-slate-100 group focus-within:ring-2 focus-within:ring-blue-500/20 focus-within:border-blue-500 transition-all">
                                    <label className="label-modern !mb-1">Total Ciclos</label>
                                    <div className="flex items-center gap-3">
                                        <div className="bg-white p-2 rounded-lg text-slate-400 shadow-sm">
                                            <Hash size={20}/>
                                        </div>
                                        <input 
                                            type="number" 
                                            name="numero_ciclos" 
                                            value={formData.numero_ciclos} 
                                            onChange={handleChange} 
                                            className="bg-transparent border-none text-xl font-bold text-slate-800 w-full focus:ring-0 p-0 placeholder:text-slate-300"
                                            min="1"
                                            placeholder="4"
                                        />
                                        <span className="text-xs font-bold text-slate-400 uppercase pr-2">Sesiones</span>
                                    </div>
                                </div>
                            </div>
                        </div>

                        {/* OBSERVACIONES */}
                        <div>
                            <label className="label-modern">Notas Clínicas Iniciales</label>
                            <textarea 
                                name="observaciones" 
                                value={formData.observaciones} 
                                onChange={handleChange}
                                rows={3}
                                className="input-modern resize-none py-3"
                                placeholder="Escribe aquí cualquier observación relevante..."
                            ></textarea>
                        </div>

                        {/* ACTIONS */}
                        <div className="pt-4 flex items-center justify-end gap-3">
                            <Link 
                                href={`/navegacion/admin/pacientes/perfil?id=${patientId}`}
                                className="px-6 py-3.5 rounded-xl font-bold text-sm text-slate-500 hover:bg-slate-50 hover:text-slate-700 transition-colors"
                            >
                                Cancelar
                            </Link>
                            <button 
                                type="submit" 
                                disabled={isLoading}
                                className="px-8 py-3.5 bg-gradient-to-r from-blue-600 to-indigo-600 text-white rounded-xl font-bold text-sm shadow-lg shadow-blue-500/30 hover:shadow-blue-500/50 hover:scale-[1.02] active:scale-[0.98] transition-all flex items-center gap-2 disabled:opacity-70 disabled:hover:scale-100"
                            >
                                {isLoading ? (
                                    <>
                                        <div className="w-4 h-4 border-2 border-white/30 border-t-white rounded-full animate-spin"/>
                                        Procesando...
                                    </>
                                ) : (
                                    <>
                                        Generar Ciclos <Sparkles size={18} fill="white" className="ml-1"/>
                                    </>
                                )}
                            </button>
                        </div>
                    </form>
                </div>
            </div>

            {/* COLUMNA DERECHA: RESUMEN PREMIUM (4 cols) */}
            <div className="lg:col-span-4 space-y-6 sticky top-8">
                
                {/* Tarjeta de Resumen "Ticket" */}
                <div className="bg-slate-900 rounded-[2rem] p-6 text-white shadow-2xl shadow-slate-900/20 relative overflow-hidden">
                    {/* Decoración de fondo */}
                    <div className="absolute top-0 right-0 w-32 h-32 bg-blue-500 rounded-full blur-[60px] opacity-20 pointer-events-none"></div>
                    <div className="absolute bottom-0 left-0 w-24 h-24 bg-indigo-500 rounded-full blur-[50px] opacity-20 pointer-events-none"></div>

                    <h3 className="font-bold text-lg mb-6 flex items-center gap-2 relative z-10">
                        <CalendarDays className="text-blue-400"/> Resumen Proyección
                    </h3>

                    <div className="space-y-6 relative z-10">
                        <div className="bg-white/5 border border-white/10 rounded-2xl p-4 backdrop-blur-sm">
                            <span className="text-[10px] uppercase font-bold text-slate-400 tracking-wider">Esquema</span>
                            <div className="font-bold text-lg mt-1 truncate">
                                {formData.nombre_esquema || "Sin definir"}
                            </div>
                            <div className="text-xs text-blue-300 font-medium mt-1">{formData.tipo}</div>
                        </div>

                        <div className="flex gap-4">
                            <div className="flex-1">
                                <span className="text-[10px] uppercase font-bold text-slate-400 tracking-wider">Duración Total</span>
                                <div className="text-2xl font-black mt-1">
                                    {formData.frecuencia_dias * formData.numero_ciclos} <span className="text-sm font-bold text-slate-500">días</span>
                                </div>
                            </div>
                            <div className="flex-1">
                                <span className="text-[10px] uppercase font-bold text-slate-400 tracking-wider">Sesiones</span>
                                <div className="text-2xl font-black mt-1">
                                    {formData.numero_ciclos}
                                </div>
                            </div>
                        </div>

                        <div className="pt-4 border-t border-white/10">
                            <span className="text-[10px] uppercase font-bold text-slate-400 tracking-wider">Finalización Estimada</span>
                            <div className="text-xl font-bold text-emerald-400 mt-1 flex items-center gap-2">
                                <CheckCircle2 size={18}/> {fechaFinEstimada()}
                            </div>
                        </div>
                    </div>
                </div>

                {/* Nota pequeña */}
                <div className="bg-white p-4 rounded-2xl border border-slate-100 shadow-sm text-xs text-slate-500 leading-relaxed text-center">
                    <p>Recuerda que estas son proyecciones estimadas. Las fechas reales se ajustarán según la evolución clínica del paciente.</p>
                </div>
            </div>

        </div>
      </div>

      <style jsx>{`
        .label-modern {
            display: block;
            font-size: 0.75rem;
            font-weight: 800;
            text-transform: uppercase;
            color: #64748b; /* slate-500 */
            margin-bottom: 0.75rem;
            letter-spacing: 0.05em;
        }
        .input-modern {
            width: 100%;
            padding: 0.875rem 1.25rem;
            background-color: #f8fafc; /* slate-50 */
            border: 1px solid #e2e8f0; /* slate-200 */
            border-radius: 1rem;
            font-size: 0.95rem;
            font-weight: 600;
            color: #1e293b; /* slate-800 */
            transition: all 0.2s cubic-bezier(0.4, 0, 0.2, 1);
            outline: none;
        }
        .input-modern:focus {
            background-color: white;
            border-color: #3b82f6;
            box-shadow: 0 4px 12px rgba(59, 130, 246, 0.08);
            transform: translateY(-1px);
        }
        .input-modern::placeholder {
            color: #cbd5e1;
            font-weight: 500;
        }
      `}</style>
    </div>
  );
}

export default function Page() {
  return (
    <Suspense fallback={<div className="h-screen flex items-center justify-center text-slate-400 font-bold uppercase text-xs tracking-widest">Cargando interfaz...</div>}>
      <NuevoTratamientoContent />
    </Suspense>
  );
}
